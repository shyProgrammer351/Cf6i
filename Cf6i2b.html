코드 빈칸 채우기

#include <stdio.h> 					 
 
int main() { 
	int row = 0, column = 9; 
	char direction='w'; 
 
	printf(" [44m                  [0m \n" ); 
	( ++row < 8 ) { 
		printf(" [44m  [0m                [44m  [0m \n" ); 
	} 
	printf(" [44m                  [0m \n" ); 
	 
	( (direction=='W')||(direction=='A')||(direction=='S')||(direction=='D') 
	     ||(direction=='w')||(direction=='a')||(direction=='s')||(direction=='d') ) { 
		printf("[%d;%df  ",row,column); 
		switch( direction ) { 
			case 'w': case 'W':  
				row = ( row <= 2 ) ? 2 : row-1; 
				break; 
			case 'a': case 'A':  
				column = ( column <= 2 ) ? 2 : column-1; 
				break; 
			case 's': case 'S':  
				row = ( row >= 8 ) ? 8 : row+1; 
				break; 
			case 'd': case 'D':  
				column = ( column >= 15 ) ? 15 : column+1; 
				break; 
		} 
		printf(" [%d;%df옷",row,column); 
 
		printf(" [10;1fWASD와 [Enter]를 입력하세요:          [10;29f" ); 
		scanf("\n%c",&direction); 
	}  
 
	return 0; 
}
