문제

캐릭터를 WASD키를 이용하여 상하좌우로 이동시키는 문제입니다 실행순서를 클릭하세요 
                  
                  
                  
                  
                  
                  
                  
        옷        
                  
WASD와 [Enter]를 입력하세요: a
                  
                  
                  
                  
                  
                  
                  
       옷         
                  
WASD와 [Enter]를 입력하세요: A
                  
                  
                  
                  
                  
                  
                  
      옷          
                  
WASD와 [Enter]를 입력하세요: q

프로그램 코드

	#include <stdio.h>


1	int main() {

		int row = 0, column = 9;

2		char direction='w';


3		printf(" \033[44m                  \033[0m \n" );

4		while( ++row < 8 ) {

5			printf(" \033[44m  \033[0m                \033[44m  \033[0m \n" );

		}

6		printf(" \033[44m                  \033[0m \n" );

		

7		while( (direction=='W')||(direction=='A')||(direction=='S')||(direction=='D')

		     ||(direction=='w')||(direction=='a')||(direction=='s')||(direction=='d') ) {

8			printf("\033[%d;%df  ",row,column);

9			switch( direction ) {

10				case 'w': case 'W':

11					row = ( row <= 2 ) ? 2 : row-1;

12					break;

13				case 'a': case 'A':

14					column = ( column <= 2 ) ? 2 : column-1;

15					break;

16				case 's': case 'S':

17					row = ( row >= 8 ) ? 8 : row+1;

18					break;

19				case 'd': case 'D':

20					column = ( column >= 15 ) ? 15 : column+1;

21					break;

			}

22			printf(" \033[%d;%df옷",row,column);


23			printf(" \033[10;1fWASD와 [Enter]를 입력하세요:          \033[10;29f" );

24			scanf("\n%c",&direction);

		}


25		return 0;

	}
