문제

캐릭터를 WASD키를 이용하여 상하좌우로 이동시키는 문제입니다 실행순서를 클릭하세요 
                  
                  
                  
                  
        옷        
                  
                  
                  
                  
WASD와 [Enter]를 입력하세요: a
                  
                  
                  
                  
       옷         
                  
                  
                  
                  
WASD와 [Enter]를 입력하세요: A
                  
                  
                  
                  
      옷          
                  
                  
                  
                  
WASD와 [Enter]를 입력하세요: q

프로그램 코드

	#include <stdio.h> 					


1	int main() {

		int row = 5, column = 9;

2		char direction='\0';


3		printf(" \033[44m                  \033[0m \n" );

4		printf(" \033[44m  \033[0m                \033[44m  \033[0m \n" );

5		printf(" \033[44m  \033[0m                \033[44m  \033[0m \n" );

6		printf(" \033[44m  \033[0m                \033[44m  \033[0m \n" );

7		printf(" \033[44m  \033[0m                \033[44m  \033[0m \n" );

8		printf(" \033[44m  \033[0m                \033[44m  \033[0m \n" );

9		printf(" \033[44m  \033[0m                \033[44m  \033[0m \n" );

10		printf(" \033[44m  \033[0m                \033[44m  \033[0m \n" );

11		printf(" \033[44m                  \033[0m \n" );

		

12		while ( ( direction != 'q' ) && ( direction != 'Q' ) ) {

13			printf("\033[%d;%df  ",row,column);

14			switch( direction ) {

15				case 'w': case 'W': 

16					row = ( row <= 2 ) ? 2 : row-1;

17					break;

18				case 'a': case 'A': 

19					column = ( column <= 2 ) ? 2 : column-1;

20					break;

21				case 's': case 'S': 

22					row = ( row >= 8 ) ? 8 : row+1;

23					break;

24				case 'd': case 'D': 

25					column = ( column >= 15 ) ? 15 : column+1;

26					break;

			}

27			printf(" \033[%d;%df옷",row,column);


28			printf(" \033[10;1fWASD 또는 Q와 [Enter]를 입력하세요:          \033[10;38f" );

29			scanf("\n%c",&direction);

		} 


30		return 0;

	}
