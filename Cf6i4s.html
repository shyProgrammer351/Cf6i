문제

A양과 B군이 가위바위보를 선택하면 승패를 출력하는 문제입니다 실행순서를 클릭하세요 
A양, 가위(1), 바위(2), 보(3) 중 하나를 숫자로 입력하세요: 2
바위를 냈습니다.
B군, 가위(1), 바위(2), 보(3) 중 하나를 숫자로 입력하세요: 1
가위를 냈습니다.
A양이 이겼습니다.

프로그램 코드

	#include <stdio.h>


1	int main() {

3		printf( "A양, 가위(1), 바위(2), 보(3) 중 하나를 숫자로 입력하세요: " );

		int playerA = 0;

4		scanf( "%d", &playerA );

5		switch( playerA ) {

6		case 1:

7			printf( "가위를 냈습니다.\n" );

8			break;

10			printf( "바위를 냈습니다.\n" );

11			break;

12		case 3:

13			printf( "보를 냈습니다.\n" );

		}


14		printf( "B군, 가위(1), 바위(2), 보(3) 중 하나를 숫자로 입력하세요: " );

		int playerB = 0;

15		scanf( "%d", &playerB );

16		switch( playerB ) {

17		case 1:

18			printf( "가위를 냈습니다.\n" );

19			break;

20		case 2:

21			printf( "바위를 냈습니다.\n" );

22			break;

23		case 3:

24			printf( "보를 냈습니다.\n" );

		}


25		if ( playerA == playerB ) {

26			printf( "\nA양과 B군이 비겼습니다.\n" );

		}

27		else if ( ( playerA == 1 ) && ( playerB == 3 ) ) {

28			printf( "\nA양이 이겼습니다.\n" );

		}

29		else if ( ( playerA == 2 ) && ( playerB == 1 ) ) {

30			printf( "\nA양이 이겼습니다.\n" );

		}

31		else if ( ( playerA == 3 ) && ( playerB == 2 ) ) {

32			printf( "\nA양이 이겼습니다.\n" );

		}

33		else {

34			printf( "\nB군이 이겼습니다.\n" );
