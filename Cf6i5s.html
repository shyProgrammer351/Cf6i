문제

5번의 기회 내에 사용자가 임의의 숫자를 맞출 수 있도록, 그 숫자와 사용자 입력을 비교하여 Up 또는 Down을 출력하는 문제입니다 실행순서를 클릭하세요 
OOOOO
1~100사이의 숫자를 입력하세요: 1000
제한범위를 벗어났습니다 
OOOOO
1~100사이의 숫자를 입력하세요: 50
Up!
XOOOO 
1~100사이의 숫자를 입력하세요: 75
Sucess!!

프로그램 코드

	#include <stdio.h>

	#include <stdlib.h>

	#include <time.h>


1	int main() {

2		int number = 0, user = 0, chance = 5;


3		srand( time( NULL ) );

4		number = rand() % 100 + 1;


5		printf( "OOOOO" );

6		while( chance > 0 ) {

7			printf( " \033[2;1f1~100사이의 숫자를 입력하세요:       \033[2;32f" );

8			scanf( "%d", &user );


9			if ( ( user < 1 ) || ( 100 < user ) ) {

10				printf( "제한범위를 벗어났습니다" );

11				continue;

			}

12			else if ( number == user ) {

13				printf( "      Success!!             " );

14				break;

			}

15			else if ( number > user ) {

16				printf( "Up!                        " );

			}

17			else if ( number < user ) {

18				printf( "Down!                      " );

			}

			

19			printf( " \033[1;%dfX", 6 - (chance--) );

		}

		

20		printf( " \033[2;1f                                 \n      " );


21		return 0;

	}


	#include <stdio.h>

	#include <stdlib.h>

	#include <time.h>


	int main() {

		int number = 0, user = 0, chance = 6;


		srand( time( NULL ) );

		number = rand() % 100 + 1;


		while( --chance > 0 ) {

		    printf( "기회는 %d번 남았습니다\n", chance );

		    

			printf( "1~100사이의 숫자를 입력하세요:" );

			scanf( "%d", &user );


			if ( ( user < 1 ) || ( 100 < user ) ) {

				printf( "제한범위를 벗어났습니다\n" );

				chance++;

				continue;

			}

			else if ( number == user ) {

				printf( "Success!!\n" );

				break;

			}

			else if ( number > user ) {

				printf( "Up!\n" );

			}

			else if ( number < user ) {

				printf( "Down!\n" );

			}

		}


		return 0;

	}
